# Trade Management API — sample curl commands (base URL: http://localhost:8080/project)
# Use curl.exe on Windows PowerShell if `curl` is aliased to Invoke-WebRequest.

# --- Create trade ---
curl.exe -X POST "http://localhost:8080/project/api/trades" -H "Content-Type: application/json" -d "{\"tradeId\":\"T-2026-0001\",\"symbol\":\"AAPL\",\"side\":\"BUY\",\"quantity\":100.5,\"price\":182.35,\"status\":\"NEW\",\"eventTimestamp\":\"2026-04-03T14:30:00\",\"sourceSystem\":\"OMS-1\"}"

# --- List all trades ---
curl.exe -X GET "http://localhost:8080/project/api/trades"

# --- Get by database id (replace 1) ---
curl.exe -X GET "http://localhost:8080/project/api/trades/1"

# --- Get by business tradeId ---
curl.exe -X GET "http://localhost:8080/project/api/trades/business/T-2026-0001"

# --- Update trade (replace 1) ---
curl.exe -X PUT "http://localhost:8080/project/api/trades/1" -H "Content-Type: application/json" -d "{\"tradeId\":\"T-2026-0001\",\"symbol\":\"AAPL\",\"side\":\"SELL\",\"quantity\":50,\"price\":190,\"status\":\"AMENDED\",\"eventTimestamp\":\"2026-04-03T15:00:00\",\"sourceSystem\":\"OMS-1\"}"

# --- Delete trade (replace 1) ---
curl.exe -X DELETE "http://localhost:8080/project/api/trades/1"
